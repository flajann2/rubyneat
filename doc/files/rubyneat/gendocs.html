#!/bin/bash # Generate docs for the Market Analysis System. # You must be logged into the embracer project!!

docroot=$devnow

pro cd $devnow rm -fr rdoc

rdoc -Ua –main docs.rb -o rdoc -t “Ruby NeuralEvolution of Augmented Topologies” $docroot rdoc -Ua –ri –main docs.rb -t “Ruby NeuralEvolution of Augmented Topologies” $docroot
